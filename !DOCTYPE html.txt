  <script>
    const products = [
      { name: "T-shirt futuriste", price: 25, image: "https://picsum.photos/200?random=1" },
      { name: "Casquette intelligente", price: 30, image: "https://picsum.photos/200?random=2" },
      { name: "Chaussures connect√©es", price: 50, image: "https://picsum.photos/200?random=3" }
    ];

    const productList = document.getElementById("productList");
    const cartItems = document.getElementById("cartItems");
    const totalEl = document.getElementById("total");

    let total = 0;

    products.forEach(product => {
      const div = document.createElement("div");
      div.classList.add("product");
      div.innerHTML = `
        <img src="${product.image}" alt="${product.name}">
        <h3>${product.name}</h3>
        <p>${product.price} $</p>
        <button>Ajouter au panier</button>
      `;
      div.querySelector("button").addEventListener("click", () => {
        const li = document.createElement("li");
        li.textContent = `${product.name} - ${product.price} $`;
        cartItems.appendChild(li);
        total += product.price;
        totalEl.textContent = total;
      });
      productList.appendChild(div);
    });
  </script>
